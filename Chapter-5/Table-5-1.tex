    \nonlineartype{幂函数}
    & \nonlinearexp{$$ y = ax^b $$}
    & \nonlineartrans{$$ \begin{matrix}
        y' = \ln y, \beta_0 = \ln a, x' = \ln x, \beta_1 = b \\
        \ln y = \ln a + b \ln x
    \end{matrix} $$}
    & \nonlinearlinearexp{$$ y' = \beta_0 + \beta_1 x' $$}

    & \nonlinearexp{$$ y = ae^{bx} $$}
    & \nonlineartrans{$$ \begin{matrix}
        y' = \ln y, a' = \ln a \\
        \ln y = \ln a + bx
    \end{matrix} $$}
    & \nonlinearlinearexp{$$ y' = a' + bx $$}
    \\\hline

    \nonlineartype{倒指数曲线}

    & \nonlineartrans{$$ \begin{matrix}
        y' = \ln y, a' = \ln a, x' = \ddfrac{1}{x} \\
        \ln y = \ln a + \ddfrac{b}{x}
    \end{matrix} $$}
    & \nonlinearlinearexp{$$ y' = a' + bx' $$}
    \\\hline
